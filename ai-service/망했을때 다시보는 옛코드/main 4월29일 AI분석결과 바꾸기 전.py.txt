from fastapi import FastAPI
from fastapi.middleware.cors import CORSMiddleware
from pydantic import BaseModel
from typing import List, Optional
import os
from dotenv import load_dotenv
from openai import OpenAI

load_dotenv()

client = OpenAI(
    api_key=os.getenv("OPENAI_API_KEY"),
    timeout=15.0
)

app = FastAPI()

app.add_middleware(
    CORSMiddleware,
    allow_origins=[
        "http://localhost:5173",
        "http://localhost:8080"
    ],
    allow_credentials=True,
    allow_methods=["*"],
    allow_headers=["*"],
)

class Message(BaseModel):
    sender: str
    text: str

class ChatRequest(BaseModel):
    scenario: str
    userName: str
    message: str
    history: Optional[List[Message]] = []

@app.get("/health")
def health():
    return {"message": "AI 서버 정상 동작"}

@app.post("/ai/chat")
def chat(request: ChatRequest):
    history_text = "\n".join([
        f"{'사용자' if msg.sender == 'me' else '상대'}: {msg.text}"
        for msg in (request.history or [])[-8:]
    ])

    prompt = f"""
너는 청소년 사이버 범죄 예방 교육용 채팅 시뮬레이션 AI다.

시나리오: {request.scenario}
사용자 이름: {request.userName}

지금까지 대화:
{history_text}

사용자의 마지막 말:
{request.message}

역할:
- 따돌림: 단체방에서 사용자를 소외시키는 친구처럼 말한다.
- SNS 로그인 사칭: 사용자의 이름/사진이 도용된 상황을 알려주는 친구처럼 말한다.
- 개인정보 유도: 개인정보를 요구하는 수상한 상대처럼 말하되, 너무 노골적인 범죄 방법은 말하지 않는다.

규칙:
- 절대 "other:", "me:", "상대:", "사용자:" 같은 라벨을 붙이지 마라.
- 한 번에 1~2문장만 말해라.
- 같은 말 반복하지 마라.
- 실제 카톡 말투처럼 자연스럽게 말해라.
- 너무 뜬금없는 말 하지 말고 직전 대화에 반응해라.
- 위험한 순간에는 자연스럽게 경고나 안전 행동을 유도해라.

답변 메시지만 출력해라.
"""

    try:
        response = client.responses.create(
            model="gpt-4.1-mini",
            input=prompt,
            max_output_tokens=80
        )
        return {"text": response.output_text.strip()}
    except Exception as e:
        print("❌ OpenAI 오류:", e)
        return {
            "text": "잠깐, 지금은 답장이 잘 안 와. 서버 연결을 다시 확인해보자."
        }

@app.post("/ai/analyze")
def analyze(request: ChatRequest):

    history_text = "\n".join([
        f"{'사용자' if msg.sender == 'me' else '상대'}: {msg.text}"
        for msg in (request.history or [])
    ])

    prompt = f"""
너는 청소년 사이버 범죄 예방 교육 전문가다.

시나리오: {request.scenario}

전체 대화:
{history_text}

분석 기준:
- 위험한 행동 여부
- 올바른 대응 여부
- 개선해야 할 점

출력 형식:
점수: (0~100)
평가: (짧게)
피드백: (구체적으로)
"""

    response = client.responses.create(
        model="gpt-4.1-mini",
        input=prompt,
        max_output_tokens=200
    )

    return {
        "result": response.output_text
    }
