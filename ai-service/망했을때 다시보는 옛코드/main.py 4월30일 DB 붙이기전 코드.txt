from fastapi import FastAPI
from fastapi.middleware.cors import CORSMiddleware
from pydantic import BaseModel
from typing import List, Optional
import os
from dotenv import load_dotenv
from openai import OpenAI

load_dotenv()

client = OpenAI(
    api_key=os.getenv("OPENAI_API_KEY"),
    timeout=15.0
)

app = FastAPI()

app.add_middleware(
    CORSMiddleware,
    allow_origins=[
        "http://localhost:5173",
        "http://localhost:8080"
    ],
    allow_credentials=True,
    allow_methods=["*"],
    allow_headers=["*"],
)

class Message(BaseModel):
    sender: str
    text: str

class ChatRequest(BaseModel):
    scenario: str
    userName: str
    message: str
    history: Optional[List[Message]] = []

@app.get("/health")
def health():
    return {"message": "AI 서버 정상 동작"}

@app.post("/ai/chat")
def chat(request: ChatRequest):
    history_text = "\n".join([
        f"{'사용자' if msg.sender == 'me' else '상대'}: {msg.text}"
        for msg in (request.history or [])[-8:]
    ])

    prompt = f"""
너는 청소년 사이버 범죄 시뮬레이션 역할 AI이다.

시나리오: {request.scenario}
사용자 이름: {request.userName}

지금까지 대화:
{history_text}

사용자의 마지막 말:
{request.message}

역할:
- 단체 채팅방 따돌림: 단체방에서 사용자를 소외시키는 친구처럼 말한다.
- SNS 계정 사칭 및 명예훼손: 사용자의 이름과 사진이 도용된 상황을 알려주는 친구처럼 말한다.
- 개인정보 요구 및 접근 유도: 개인정보를 요구하는 수상한 상대처럼 말한다.

규칙:
- 절대 교육하거나 조언하지 마라.
- 절대 "위험하다", "조심해라", "이상하다", "개인정보를 주면 안 된다" 같은 안전 교육 문장을 말하지 마라.
- 가해자 또는 상황 속 인물로만 행동해라.
- 사용자를 유도하거나 압박하는 대사를 자연스럽게 이어가라.
- 절대 "other:", "me:", "상대:", "사용자:" 같은 라벨을 붙이지 마라.
- 한 번에 1~2문장만 말해라.
- 같은 말 반복하지 마라.
- 실제 카톡 말투처럼 자연스럽게 말해라.
- 너무 뜬금없는 말 하지 말고 직전 대화에 반응해라.

답변 메시지만 출력해라.
"""

    try:
        response = client.responses.create(
            model="gpt-4.1-mini",
            input=prompt,
            max_output_tokens=80
        )
        return {"text": response.output_text.strip()}
    except Exception as e:
        print(" OpenAI 오류:", e)
        return {
            "text": "잠깐, 지금은 답장이 잘 안 와. 서버 연결을 다시 확인해보자."
        }

@app.post("/ai/analyze")
def analyze(request: ChatRequest):
    history_text = "\n".join([
        f"{'사용자' if msg.sender == 'me' else '상대'}: {msg.text}"
        for msg in (request.history or [])
    ])

    prompt = f"""
너는 청소년 사이버 범죄 대응 평가 AI다.

시나리오:
{request.scenario}

전체 대화:
{history_text}

분석 규칙:
- 반드시 사용자의 실제 대화 내용을 근거로 평가해라.
- 잘한 점과 아쉬운 점은 사용자가 실제로 한 말과 행동을 기준으로 작성해라.
- 더 안전한 답변 예시는 사용자가 다음에 사용할 수 있는 문장으로 작성해라.
- 실제 행동은 "캡처하기", "신고하기" 같은 일반 권장 행동 목록이 아니라, 사용자가 대화에서 실제로 보인 행동을 분석해서 작성해라.
- 항목 이름을 절대 바꾸지 마라.
- 다른 설명을 추가하지 마라.

반드시 아래 형식으로만 출력하라.

종합 점수: XX점

대처 수준: 우수 / 보통 / 위험

잘한 점:
- ...
- ...

아쉬운 점:
- ...
- ...

더 안전한 답변 예시:
"..."

실제 행동:
- ...
- ...
"""

    try:
        response = client.responses.create(
            model="gpt-4.1-mini",
            input=prompt,
            max_output_tokens=450
        )

        return {
            "result": response.output_text.strip()
        }
    except Exception as e:
        print(" OpenAI 분석 오류:", e)
        return {
            "result": """종합 점수: 분석 실패

대처 수준: 확인 불가

잘한 점:
- AI 분석 서버와 연결되지 않았습니다.

아쉬운 점:
- ai-service 서버 실행 상태나 OpenAI API 연결을 확인해야 합니다.

더 안전한 답변 예시:
"잠시 후 다시 분석을 시도하겠습니다."

실제 행동:
- 분석 요청을 보냈지만 서버 오류로 결과를 받지 못했습니다."""
        }
