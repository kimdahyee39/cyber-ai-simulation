from fastapi import FastAPI
from fastapi.middleware.cors import CORSMiddleware
from pydantic import BaseModel
from typing import List, Optional

app = FastAPI()

app.add_middleware(
    CORSMiddleware,
    allow_origins=["http://localhost:5173", "http://localhost:8080"],
    allow_credentials=True,
    allow_methods=["*"],
    allow_headers=["*"],
)

class Message(BaseModel):
    sender: str
    text: str

class ChatRequest(BaseModel):
    scenario: str
    userName: str
    message: str
    history: Optional[List[Message]] = []

@app.get("/health")
def health():
    return {"message": "AI 서버 정상 동작"}

@app.post("/ai/chat")
def chat(request: ChatRequest):
    turn = len(request.history or [])
    name = request.userName

    if request.scenario == "따돌림":
        replies = [
            f"{name}, 너만 빼고 우리끼리 얘기하기로 했어.",
            "아까 단톡방에서 너 얘기 나왔는데 봤어?",
            f"{name}, 왜 갑자기 조용해? 기분 나쁜 거야?",
            "이런 상황이면 그냥 캡처해 두고 선생님께 말하는 게 좋을 수도 있어."
        ]

    elif request.scenario == "SNS 로그인 사칭":
        replies = [
            {
                "text": f"{name}, 이 계정 너 맞아? 누가 네 사진으로 이상한 계정을 만들었대.",
                "imageUrl": "/fake-profile.png"
            },
            "프로필 사진도 네 사진이랑 비슷해 보여.",
            "DM으로 친구들한테 이상한 링크도 보내고 있대.",
            "이건 바로 신고하고 비밀번호도 바꾸는 게 좋아."
        ]

    elif request.scenario == "개인정보 유도":
        replies = [
            f"{name}, 친해지고 싶어서 그러는데 전화번호 알려줄래?",
            "학교랑 반도 알려주면 더 빨리 찾을 수 있을 것 같은데.",
            "주소 알려주면 선물 보내줄게.",
            "개인정보는 절대 보내면 안 돼. 캡처하고 차단하는 게 좋아."
        ]

    else:
        replies = [
            f"{name}, 지금 상황을 잘 생각해보고 답장해 봐."
        ]

    reply = replies[turn % len(replies)]

    if isinstance(reply, dict):
        return reply

    return {"text": reply}
